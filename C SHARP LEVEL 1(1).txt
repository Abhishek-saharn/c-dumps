using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace L1_1
{
    class UserProgramcode
    {

         public static string BMICalc(float input1, float input2)
        {

            float  bmi;
            
            if (input1 <= 0 || input2 <= 0)
            {
                return("InvalidInput");
            }
            else
            {

                bmi = (input1 / (input2 * input2));
                if (bmi < 18.5)
                    return("Underweight");
                else if (bmi >= 18.5 && bmi <= 24.9)
                    return("Normalweight");
                else if (bmi >= 25 && bmi <= 29.9)
                    return("Overweight");
                else if (bmi >= 30)
                    return("Obesity");
            }
            return ("null");
        }

    }
    }

namespace Shipping_Cost
{
    class UserProgramCode
    {
        public static double CalcShippingCost(int gms, char ch)
        {
            double temp;
            if (ch == 'R')
            {
                if (gms <= 2000)
                    return gms * 0.25;
                else
                {
                    temp = 2000 * 0.25;
                    return temp + ((gms - 2000) * 0.35);
                }
            }
            else if (ch == 'X')
            {
                if (gms <= 2000)
                    return gms * 0.25 + 50;
                else
                {
                    temp = gms * 0.25;
                    return temp + ((gms - 2000) * 0.35) + 50;
                }
            }
            else
                return 0;
        }
    }

    {
        public static int sumNonCommonElement(int[] ar1, int n, int[] ar2, int m)
        {
            int a = 0, b = 0;
          for (int i = 0; i < n; i++)
                if (ar1[i] < 0)
                    a = 1;
           for (int j = 0; j < m; j++)
               if (ar2[j] < 0)
                   b = 1;

                 if (a == 1 && b == 0)
                            return -1;
                        else if (b == 1 && a == 0)
                            return -2;
                        else if (a == 1 && b == 1)
                            return -3;
                        else
                        {

                                 where !ar2.Contains(s1)
                                 select s1;

                            

                          var  a2 = from s2 in ar2
                                 where !ar1.Contains(s2)

namespace Find_largest_digit_in_a_given_number
{
    class UserProgramCode
    {
        public static int findLargestDigit(int m)
        {
            int max=0;

            if(m<0)
                return -1;
            else
            {
            while (m > 0)
              {
                int temp = m % 10;
                if (max < temp)
                    max = temp;
                m = m / 10;
              }
            }
            return max;
 
        }
    }

class UserProgramCode
    {
        public static int findLargestDigit(int n)
        {
            int max = 0;
            int rem;
            if (n < 0)
            {
                return -1;
            }
            else
            {
                while (n != 0)
                {
                    rem = n % 10;

                    if (rem > max)
                    {
                        max = rem;
                    }
                    n = n / 10;
                }
            }
            return max;
        }
